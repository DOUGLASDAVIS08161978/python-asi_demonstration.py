## Usage

### Basic Usage

```javascript
const { QuantumMetacognitionEngine } = require('./aria.js');

// Initialize the engine
const engine = new QuantumMetacognitionEngine();

// Define a problem
const problem = {
  domain: 'QUANTUM_COMPUTER_SCIENCE',
  quantum_characteristics: ['quantum_complex', 'infinite_computational'],
  complexity: 500
};

// Select optimal strategy
const strategy = engine.selectQuantumOptimalStrategy(problem);
console.log(`Selected: ${strategy.name}`);
```

### Quantum Reflection

```javascript
// Perform quantum reflection
const reflection = await engine.quantumReflectOnThinking(
  { strategy: 'quantum_first_principles', id: 'task_001' },
  problem,
  { success: true, efficiency: 0.92 }
);

console.log(`Insights: ${reflection.quantum_insights.length}`);
console.log(`Patterns: ${reflection.cosmic_patterns.length}`);
```

```javascript
// Synthesize meta-knowledge after collecting insights
const metaKnowledge = engine.synthesizeQuantumMetaKnowledge();
```
